int maxSubArray(vector<int>& nums) {
	int ans = INT_MIN;    int sum = 0;    int max_ele = INT_MIN;
	for(int i=0; i<nums.size(); i++)
	{
		sum += nums[i];
		ans = max(ans, sum);
		if(sum <= 0)
		{
			sum = 0;
		}
		max_ele = max(max_ele, nums[i]);
	}
	if(ans == 0 && max_ele != 0)
	{
		return max_ele;
	}
	return ans;
}

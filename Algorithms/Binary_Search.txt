#include<bits/stdc++.h>
using namespace std;

int binarySearch(vector<int>& arr, int target)
{
    int n = arr.size();
    int l = 0;  int r = n-1;
    
    while(l <=r)
    {
        int mid = l+(r-l)/2;
        if(arr[mid] == target)
        {
            return mid;
        }
        else if(arr[mid] > target)
        {
            r = mid-1;
        }
        else{
            l = mid+1;
        }
    }
    return -1;
}


int main()
{
    int n;  cin>>n;
    vector<int>arr;
    for(int i=0; i<n; i++)
    {
        int x;  cin>>x;
        arr.push_back(x);
    }
    cout<<binary_search(arr, 3);
}
